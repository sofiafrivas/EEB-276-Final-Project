pc_scores <- as.data.frame(pca$x[, 1:10])

pc_data <- pc_scores %>%

  prior(normal(0, 1), class = "b"),
  prior(normal(6.5, 2), class = "Intercept"), #mean mean_gi is ~6.5
  prior(student_t(3, 0, 2), class = "sigma"))

#run brm model

                 data = pc_data,
                 family = gaussian(),
                 prior = priors_pca,
                 chains = 4, iter = 4000, warmup = 2000,
                 cores = 4)
